6   package com.hp.autonomy.frontend.reports.powerpoint.dto;
7   
8   import java.util.List;
9   import lombok.AllArgsConstructor;
10  import lombok.Data;
11  import lombok.NoArgsConstructor;
12  
13  /**
14   * DTO to represent time-series data to plot on a xy scatterplot chart with time on the x-axis.
15   */
16  @Data
17  @NoArgsConstructor
18  @AllArgsConstructor
19  public class DategraphData implements ComposableElement {
20  
21      /** A list of timestamps as Unix epoch seconds. */
22      private long[] timestamps;
23  
24      /* One of more row data series with values corresponding with the timestamps. */
25      private List<Row> rows;
26  
27      public boolean validateInput() {
28          final int length = this.timestamps.length;
29  
30          for(Row row : rows) {
31              if (row.getValues().length != length) {
32                  return false;
33              }
34          }
35  
36          return length > 1 && !rows.isEmpty();
37      }
38  
39      /**
40       * A DTO representing a data series in a date chart.
41       */
42      @Data
43      @NoArgsConstructor
44      @AllArgsConstructor
45      public static class Row {
46  
47          /** The colour to render the series data, should be a hexadecimal string e.g. #FF0000 */
48          private String color;
49  
50          /** The label for the series data. */
51          private String label;
52  
53          /** Whether the row is rendered on the primary or secondary y-axis. */
54          private boolean secondaryAxis;
55  
56          /** List of values; should be the same length as the timestamps. */
57          private double[] values;
58      }
59  }
