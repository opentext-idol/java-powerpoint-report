6   package com.hp.autonomy.frontend.reports.powerpoint.dto;
7   
8   import lombok.Data;
9   import lombok.NoArgsConstructor;
10  
11  /**
12   * Anchor points to use when drawing a report. All units should be fractional between 0 and 1,
13   * e.g. x=0.1 means start drawing 10% from the left edge of the available space, and
14   *      w=0.5 means we only draw on half of the available width.
15   */
16  @Data
17  @NoArgsConstructor
18  public class Anchor {
19  
20      public Anchor(final double x, final double y, final double width, final double height) {
21          this.x = x;
22          this.y = y;
23          this.width = width;
24          this.height = height;
25      }
26  
27      /** The left margin of the visualization, expressed as fraction between 0 and 1 of the available space. */
28      private double x;
29  
30      /** The top margin of the visualization, expressed as fraction between 0 and 1 of the available space. */
31      private double y;
32  
33      /** The width of the visualization, expressed as fraction between 0 and 1 of the available space. */
34      private double width = 1;
35  
36      /** The height of the visualization, expressed as fraction between 0 and 1 of the available space. */
37      private double height = 1;
38  
39  }
